** Wine的基本配置

*** 配置
#+begin_src emacs-lisp
  winecfg
#+end_src
*** 创建新环境
#+begin_src shell
  export WINEPREFIX=~/wine/software winecfg
  WINEARCH=win32 WINEPREFIX=~/.wine32 winecfg # 32位
  WINEPREFIX=... LC_ALL=zh_CN.UTF-8 winecfg # 中文
#+end_src
